/**
 * Validasi.
 *
 * @param array $data
 * @param array $custom
 *
 * @return array
 */
function validasi($data, $custom = [])
{
    $validasi = [
        'nama' => 'required',
        'kode' => 'required',
    ];

    return validate($data, $validasi, $custom);
}
